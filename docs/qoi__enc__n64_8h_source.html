196static inline uint32_t qoi_get_be32(uint32_t value)
197{
198 uint8_t* bytes = (uint8_t*)&value;
199 uint32_t be_value = (uint32_t) (
200 (bytes[0] << 24) |
201 (bytes[1] << 16) |
202 (bytes[2] << 8) |
203 (bytes[3])
204 );
205
206 return be_value;
207}
208
212static inline uint32_t qoi_to_be32(uint32_t value)
213{
214 uint8_t bytes[4];
215
216 bytes[0] = (value >> 24);
217 bytes[1] = (value >> 16);
218 bytes[2] = (value >> 8);
219 bytes[3] = (value);
220
221 return *((uint32_t*)bytes);
222}

229static bool qoi_cmp_pixel(qoi_pixel_t pixel1, qoi_pixel_t pixel2, const uint8_t channels)
230{
231 if (channels < 4) /* RGB pixels have three channels; RGBA pixels have four channels for the alpha channel */
232 {
233 /* O2 optimization will mask out these alpha values using OR instruction so it compares only the colors of a pixel */
234 pixel1.alpha = 255;
235 pixel2.alpha = 255;
236 }
237
238 return pixel1.concatenated_pixel_values == pixel2.concatenated_pixel_values; /* compare pixels */
239}

327void write_qoi_header(qoi_desc_t *desc, void* dest)
328{
329 if (dest == NULL || desc == NULL) return;
330
331 uint8_t *byte = (uint8_t*)dest;
332
333 /* Write the magic characters to the file first */
334 *(uint32_t*)byte = *(uint32_t*)QOI_MAGIC;
335
336 /* Writes all the metadata information about the image to the file */
337
338 uint32_t* dimension_ptr = (uint32_t*)&byte[4];
339
340 /* Writes the width and height values of the image to QOI header which stores them in big endian */
341 dimension_ptr[0] = qoi_to_be32(desc->width);
342 dimension_ptr[1] = qoi_to_be32(desc->height);
343
344 byte[12] = desc->channels;
345 byte[13] = desc->colorspace;
346}

385static inline void qoi_enc_diff(qoi_enc_t *enc, uint8_t red_diff, uint8_t green_diff, uint8_t blue_diff)
386{
387 uint8_t tag =
388 QOI_OP_DIFF |
389 (uint8_t)(red_diff + 2) << 4 |
390 (uint8_t)(green_diff + 2) << 2 |
391 (uint8_t)(blue_diff + 2);
392
393 enc->enc_buffer[enc->buffer_offset++] = tag;
394}

398static inline void qoi_enc_luma(qoi_enc_t *enc, uint8_t green_diff, uint8_t dr_dg, uint8_t db_dg)
399{
400 enc->enc_buffer[enc->buffer_offset + 0] = QOI_OP_LUMA | (uint8_t)(green_diff + 32);
401 enc->enc_buffer[enc->buffer_offset + 1] = (uint8_t)(dr_dg + 8) << 4 | (uint8_t)(db_dg + 8);
402
403 enc->buffer_offset += 2;
404}

408static inline void qoi_enc_run(qoi_enc_t *enc)
409{
410 /* The run-length is stored with a bias of -1 */
411 uint8_t tag = QOI_OP_RUN | (enc->run - 1);
412 enc->run = 0;
413
414 enc->enc_buffer[enc->buffer_offset++] = tag;

419void qoi_encode_chunk(qoi_desc_t *desc, qoi_enc_t *enc, void *qoi_pixel_bytes)
420{
421
422 /*
423 Assume that the pixel byte order is the following below
424 bytes[0] = red;
425 bytes[1] = green;
426 bytes[2] = blue;
427 bytes[3] = alpha;
428 */
429
430 qoi_pixel_t cur_pixel = *((qoi_pixel_t*)qoi_pixel_bytes);
431
432 /* Assume an RGB pixel with three channels has an alpha value that makes pixels opaque */
433 if (desc->channels < 4)
434 cur_pixel.alpha = 255;
435
436 uint8_t index_pos = qoi_get_index_position(cur_pixel);
437 enc->pixels_written++;
438
439 /* Increment run length by 1 if pixels are the same */
440 if (qoi_cmp_pixel(cur_pixel, enc->prev_pixel, desc->channels))
441 {
442 /* Note that the runlengths 63 and 64 (b111110 and b111111) are illegal as they are
443 occupied by the QOI_OP_RGB and QOI_OP_RGBA tags. */
444 if (++enc->run >= 62 || enc->pixels_written >= enc->len)
445 {
446 qoi_enc_run(enc);
447 }
448 }
449 else
450 {
451 if (enc->run > 0)
452 {
453 /* Write opcode for because there are differences in pixels
454 The run-length is stored with a bias of -1 */
455 qoi_enc_run(enc);
456 }
457
458 /* Check if pixels exist in one of the pixel hash buffers */
459 if (qoi_cmp_pixel(enc->pix_buffer[index_pos], cur_pixel, 4))
460 {
461 qoi_enc_index(enc, index_pos);
462 }
463 else
464 {
465 enc->pix_buffer[index_pos] = cur_pixel;
466
467 /* QOI doesn't have opcodes for alpha values so check alpha values between two pixels first */
468 if (desc->channels > 3 && cur_pixel.alpha != enc->prev_pixel.alpha)
469 {
470 qoi_enc_rgba(enc, cur_pixel);
471 }
472 else
473 {
474 /* Check the difference between color values to determine opcode */
475 int8_t red_diff, green_diff, blue_diff;
476 int8_t dr_dg, db_dg;
477
478 red_diff = cur_pixel.red - enc->prev_pixel.red;
479 green_diff = cur_pixel.green - enc->prev_pixel.green;
480 blue_diff = cur_pixel.blue - enc->prev_pixel.blue;
481
482 dr_dg = red_diff - green_diff;
483 db_dg = blue_diff - green_diff;
484
485 if (
486 red_diff >= -2 && red_diff <= 1 &&
487 green_diff >= -2 && green_diff <= 1 &&
488 blue_diff >= -2 && blue_diff <= 1
489 )
490 {
491 qoi_enc_diff(enc, red_diff, green_diff, blue_diff);
492 }
493
494 else if (
495 dr_dg >= -8 && dr_dg <= 7 &&
496 green_diff >= -32 && green_diff <= 31 &&
497 db_dg >= -8 && db_dg <= 7
498 )
499 {
500 qoi_enc_luma(enc, green_diff, dr_dg, db_dg);
501 }
502
503 /* otherwise write an RGB tag containting the RGB values of a pixel */
504 else
505 {
506 qoi_enc_rgb(enc, cur_pixel);
507 }
508 }
509
510 }
511 }
512
513 /* Advance the pixel offset by one and sets the previous pixel to the current pixel */
514 enc->prev_pixel = cur_pixel;
515 enc->pixel_offset++;
516
517}

522bool qoi_enc_init(qoi_desc_t* desc, qoi_enc_t* enc)
523{
524 if (enc == NULL || desc == NULL) return false;
525
526 for (uint8_t element = 0; element < 64; element++) {
527 /* Initalize all the pixels in the buffer to zero for each channel of each pixels */
528 qoi_initalize_pixel(&enc->pix_buffer[element]);
529 }
530
531 enc->len = (uint32_t)desc->width * (uint32_t)desc->height;
532
533 enc->run = 0;
534 enc->pixel_offset = 0;
535 enc->pixels_written = 0;
536
537 /*
538 The decoder and encoder start with
539 {r: 0, g: 0, b: 0, a: 255}
540 as the previous pixel value.
541 */
542 qoi_set_pixel_rgba(&enc->prev_pixel, 0, 0, 0, 255);
543
544 return true;
545}
