208static inline uint32_t qoi_get_be32(uint32_t value)
209{
210 uint8_t* bytes = (uint8_t*)&value;
211 uint32_t be_value = (uint32_t) (
212 (bytes[0] << 24) |
213 (bytes[1] << 16) |
214 (bytes[2] << 8) |
215 (bytes[3])
216 );
217
218 return be_value;
219}
220
224static inline uint32_t qoi_to_be32(uint32_t value)
225{
226 uint8_t bytes[4];
227
228 bytes[0] = (value >> 24);
229 bytes[1] = (value >> 16);
230 bytes[2] = (value >> 8);
231 bytes[3] = (value);
232
233 return *((uint32_t*)bytes);
234}

241static bool qoi_cmp_pixel(qoi_pixel_t pixel1, qoi_pixel_t pixel2, const uint8_t channels)
242{
243 if (channels < 4) /* RGB pixels have three channels; RGBA pixels have four channels for the alpha channel */
244 {
245 /* O2 optimization will mask out these alpha values using OR instruction so it compares only the colors of a pixel */
246 pixel1.alpha = 255;
247 pixel2.alpha = 255;
248 }
249
250 return pixel1.concatenated_pixel_values == pixel2.concatenated_pixel_values; /* compare pixels */
251}

339void write_qoi_header(qoi_desc_t *desc, void* dest)
340{
341 if (dest == NULL || desc == NULL) return;
342
343 uint8_t *byte = (uint8_t*)dest;
344
345 /* Write the magic characters to the file first */
346 *(uint32_t*)byte = *(uint32_t*)QOI_MAGIC;
347
348 /* Writes all the metadata information about the image to the file */
349
350 uint32_t* dimension_ptr = (uint32_t*)&byte[4];
351
352 /* Writes the width and height values of the image to QOI header which stores them in big endian */
353 dimension_ptr[0] = qoi_to_be32(desc->width);
354 dimension_ptr[1] = qoi_to_be32(desc->height);
355
356 byte[12] = desc->channels;
357 byte[13] = desc->colorspace;
358}

397static inline void qoi_enc_diff(qoi_enc_t *enc, uint8_t red_diff, uint8_t green_diff, uint8_t blue_diff)
398{
399 uint8_t tag =
400 QOI_OP_DIFF |
401 (uint8_t)(red_diff + 2) << 4 |
402 (uint8_t)(green_diff + 2) << 2 |
403 (uint8_t)(blue_diff + 2);
404
405 enc->enc_buffer[enc->buffer_offset++] = tag;
406}

410static inline void qoi_enc_luma(qoi_enc_t *enc, uint8_t green_diff, uint8_t dr_dg, uint8_t db_dg)
411{
412 enc->enc_buffer[enc->buffer_offset + 0] = QOI_OP_LUMA | (uint8_t)(green_diff + 32);
413 enc->enc_buffer[enc->buffer_offset + 1] = (uint8_t)(dr_dg + 8) << 4 | (uint8_t)(db_dg + 8);
414
415 enc->buffer_offset += 2;
416}

420static inline void qoi_enc_run(qoi_enc_t *enc)
421{
422 /* The run-length is stored with a bias of -1 */
423 uint8_t tag = QOI_OP_RUN | (enc->run - 1);
424 enc->run = 0;
425
426 enc->enc_buffer[enc->buffer_offset++] = tag;
427}

431void qoi_encode_chunk(qoi_desc_t *desc, qoi_enc_t *enc, void *qoi_pixel_bytes)
432{
433
434 /*
435 Assume that the pixel byte order is the following below
436 bytes[0] = red;
437 bytes[1] = green;
438 bytes[2] = blue;
439 bytes[3] = alpha;
440 */
441
442 qoi_pixel_t cur_pixel = *((qoi_pixel_t*)qoi_pixel_bytes);
443
444 /* Assume an RGB pixel with three channels has an alpha value that makes pixels opaque */
445 if (desc->channels < 4)
446 cur_pixel.alpha = 255;
447
448 uint8_t index_pos = qoi_get_index_position(cur_pixel);
449 enc->pixels_written++;
450
451 /* Increment run length by 1 if pixels are the same */
452 if (qoi_cmp_pixel(cur_pixel, enc->prev_pixel, desc->channels))
453 {
454 /* Note that the runlengths 63 and 64 (b111110 and b111111) are illegal as they are
455 occupied by the QOI_OP_RGB and QOI_OP_RGBA tags. */
456 if (++enc->run >= 62 || enc->pixels_written >= enc->len)
457 {
458 qoi_enc_run(enc);
459 }
460 }
461 else
462 {
463 if (enc->run > 0)
464 {
465 /* Write opcode for because there are differences in pixels
466 The run-length is stored with a bias of -1 */
467 qoi_enc_run(enc);
468 }
469
470 /* Check if pixels exist in one of the pixel hash buffers */
471 if (qoi_cmp_pixel(enc->pix_buffer[index_pos], cur_pixel, 4))
472 {
473 qoi_enc_index(enc, index_pos);
474 }
475 else
476 {
477 enc->pix_buffer[index_pos] = cur_pixel;
478
479 /* QOI doesn't have opcodes for alpha values so check alpha values between two pixels first */
480 if (desc->channels > 3 && cur_pixel.alpha != enc->prev_pixel.alpha)
481 {
482 qoi_enc_rgba(enc, cur_pixel);
483 }
484 else
485 {
486 /* Check the difference between color values to determine opcode */
487 int8_t red_diff, green_diff, blue_diff;
488 int8_t dr_dg, db_dg;
489
490 red_diff = cur_pixel.red - enc->prev_pixel.red;
491 green_diff = cur_pixel.green - enc->prev_pixel.green;
492 blue_diff = cur_pixel.blue - enc->prev_pixel.blue;
493
494 dr_dg = red_diff - green_diff;
495 db_dg = blue_diff - green_diff;
496
497 if (
498 red_diff >= -2 && red_diff <= 1 &&
499 green_diff >= -2 && green_diff <= 1 &&
500 blue_diff >= -2 && blue_diff <= 1
501 )
502 {
503 qoi_enc_diff(enc, red_diff, green_diff, blue_diff);
504 }
505
506 else if (
507 dr_dg >= -8 && dr_dg <= 7 &&
508 green_diff >= -32 && green_diff <= 31 &&
509 db_dg >= -8 && db_dg <= 7
510 )
511 {
512 qoi_enc_luma(enc, green_diff, dr_dg, db_dg);
513 }
514
515 /* otherwise write an RGB tag containting the RGB values of a pixel */
516 else
517 {
518 qoi_enc_rgb(enc, cur_pixel);
519 }
520 }
521
522 }
523 }
524
525 /* Advance the pixel offset by one and sets the previous pixel to the current pixel */
526 enc->prev_pixel = cur_pixel;
527
528}

534bool qoi_enc_init(qoi_desc_t* desc, qoi_enc_t* enc)
535{
536 if (enc == NULL || desc == NULL) return false;
537
538 for (uint8_t element = 0; element < 64; element++) {
539 /* Initalize all the pixels in the buffer to zero for each channel of each pixels */
540 qoi_initalize_pixel(&enc->pix_buffer[element]);
541 }
542
543 enc->len = (uint32_t)desc->width * (uint32_t)desc->height;
544
545 enc->run = 0;
546 enc->pixels_written = 0;
547
548 /*
549 The decoder and encoder start with
550 {r: 0, g: 0, b: 0, a: 255}
551 as the previous pixel value.
552 */
553 qoi_set_pixel_rgba(&enc->prev_pixel, 0, 0, 0, 255);
554
555 return true;
556}
